class WhiteBoardAdapter extends ComponentAdapter {
        @Override
        public void componentResized(ComponentEvent e) {
            int new_width;
            int new_height;
            Image new_Image;
            for(ScribblePanel panel : panelList) {
                new_width = panel.getWidth();
                new_height = panel.getHeight();
                panel.penfix = new_width/panel.cur_width;
            /*reset the curImage*/

                new_Image = panel.createImage(panel.getWidth(), panel.getHeight());
                new_Image.getGraphics().drawImage(panel.curImage.getScaledInstance(new_width, new_height, Image.SCALE_SMOOTH), 0, 0, null);
                panel.curImage = new_Image;
                panel.refresh();
            }
        }
        @Override
        public void componentMoved(ComponentEvent e) {
            for(ScribblePanel panel : WhiteBoard.panelList) {
                panel.refresh();
            }
        }
    }

    public void windowActivated(WindowEvent e) {

            }
            public void windowClosed(WindowEvent e) {

            }
            public void windowClosing(WindowEvent e) {

            }
            public void windowDeactivated(WindowEvent e) {

            }
            public void windowDeiconified(WindowEvent e) {

            }
            public void windowIconified(WindowEvent e) {

            }
            public void windowOpened(WindowEvent e) {
                for(ScribblePanel panel : WhiteBoard.panelList) {
                    panel.refresh();
                }
            }
        }
